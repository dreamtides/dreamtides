```rust
use battle_state::battle::battle_state::BattleState;
use battle_state::battle::card_id::CharacterId;
use core_data::numerics::Spark;
use core_data::types::PlayerName;

const MAX_CANDIDATES: usize = 6;

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum CharacterPlacement {
    StayBack,
    MoveToFrontRank(u8),
}

#[derive(Debug, Clone)]
pub struct PositionAssignment {
    pub placements: Vec<(CharacterId, CharacterPlacement)>,
}

/// Information about an eligible character for positioning.
struct EligibleCharacter {
    id: CharacterId,
    spark: u32,
}

/// Information about an opponent front-rank threat.
struct OpponentThreat {
    column: u8,
    spark: u32,
}

/// Generates up to MAX_CANDIDATES position assignments for the given
/// board state. Returns an empty vec if no characters are eligible for
/// positioning.
pub fn generate(
    battle: &BattleState,
    player: PlayerName,
) -> Vec<PositionAssignment> {
    let eligible = eligible_characters(battle, player);
    if eligible.is_empty() {
        return vec![];
    }

    let opponent = player.opponent();
    let opp_front_threats = opponent_front_threats(battle, opponent);
    let opp_back_max_spark = opponent_back_max_spark(battle, opponent);
    let attack_column = find_attack_column(battle, player);

    let mut candidates = Vec::with_capacity(MAX_CANDIDATES);

    // 1. Always include "hold all back"
    candidates.push(hold_all(&eligible));

    // 2. Efficient blocking (assign cheapest winner per opponent)
    if let Some(assignment) = efficient_blocking(&eligible, &opp_front_threats) {
        candidates.push(assignment);
    }

    // 3. Chump blocking (assign weakest characters to strongest opponents)
    if let Some(assignment) = chump_blocking(&eligible, &opp_front_threats) {
        // Only add if it differs from efficient blocking
        if !candidates.iter().any(|c| same_placements(c, &assignment)) {
            candidates.push(assignment);
        }
    }

    // 4. Attack-focused (only profitable attacks)
    if let Some(col) = attack_column {
        if let Some(assignment) = attack_focused(&eligible, col, opp_back_max_spark) {
            candidates.push(assignment);
        }
    }

    // 5. Mixed: block highest threat + attack with best remaining
    if !opp_front_threats.is_empty() {
        if let Some(assignment) =
            mixed_block_and_attack(&eligible, &opp_front_threats, attack_column, opp_back_max_spark)
        {
            if !candidates.iter().any(|c| same_placements(c, &assignment)) {
                candidates.push(assignment);
            }
        }
    }

    // 6. Chump-block-all: chump block as many threats as possible
    if opp_front_threats.len() > 1 {
        if let Some(assignment) = chump_block_all(&eligible, &opp_front_threats) {
            if !candidates.iter().any(|c| same_placements(c, &assignment)) {
                candidates.push(assignment);
            }
        }
    }

    // Trim to MAX_CANDIDATES if we somehow generated more
    candidates.truncate(MAX_CANDIDATES);
    candidates
}

/// Selects the single best assignment by heuristic score, for use in
/// rollouts and opponent modeling where we want a quick reasonable
/// decision.
pub fn best_assignment(
    battle: &BattleState,
    player: PlayerName,
) -> Option<PositionAssignment> {
    let candidates = generate(battle, player);
    if candidates.is_empty() {
        return None;
    }

    let opponent = player.opponent();
    let opp_back_max_spark = opponent_back_max_spark(battle, opponent);

    candidates
        .into_iter()
        .max_by_key(|c| score_assignment(battle, player, c, opp_back_max_spark))
}

/// Fast heuristic score for ranking assignments.
///
/// Score = points_prevented - spark_lost + attack_points * discount
fn score_assignment(
    battle: &BattleState,
    player: PlayerName,
    assignment: &PositionAssignment,
    opp_back_max_spark: u32,
) -> i64 {
    let opponent = player.opponent();
    let mut points_prevented: i64 = 0;
    let mut spark_lost: i64 = 0;
    let mut attack_points: i64 = 0;

    for &(char_id, placement) in &assignment.placements {
        let own_spark = battle.cards.spark(player, char_id).unwrap_or_default().0 as i64;
        match placement {
            CharacterPlacement::StayBack => {}
            CharacterPlacement::MoveToFrontRank(col) => {
                let opp_char = battle.cards.battlefield(opponent).front[col as usize];
                if let Some(opp_id) = opp_char {
                    // Blocking
                    let opp_spark =
                        battle.cards.spark(opponent, opp_id).unwrap_or_default().0 as i64;
                    points_prevented += opp_spark;
                    if opp_spark > own_spark {
                        spark_lost += own_spark; // We lose our character
                    } else if opp_spark == own_spark {
                        spark_lost += own_spark; // Trade
                    }
                    // If we win the matchup (own_spark > opp_spark), no spark lost
                } else {
                    // Attacking in empty column
                    // Discount by opponent's ability to block next turn
                    let discount = if opp_back_max_spark >= own_spark as u32 {
                        // Opponent can likely block this, low expected value
                        25
                    } else {
                        75
                    };
                    attack_points += own_spark * discount / 100;
                }
            }
        }
    }

    points_prevented - spark_lost + attack_points
}

fn hold_all(eligible: &[EligibleCharacter]) -> PositionAssignment {
    PositionAssignment {
        placements: eligible.iter().map(|c| (c.id, CharacterPlacement::StayBack)).collect(),
    }
}

/// Assign cheapest character that wins each matchup, highest threat first.
fn efficient_blocking(
    eligible: &[EligibleCharacter],
    threats: &[OpponentThreat],
) -> Option<PositionAssignment> {
    if threats.is_empty() {
        return None;
    }

    let mut sorted_threats: Vec<&OpponentThreat> = threats.iter().collect();
    sorted_threats.sort_by(|a, b| b.spark.cmp(&a.spark));

    let mut used = vec![false; eligible.len()];
    let mut placements: Vec<(CharacterId, CharacterPlacement)> = Vec::new();

    for threat in &sorted_threats {
        // Find cheapest character that beats this threat
        let best_idx = eligible
            .iter()
            .enumerate()
            .filter(|(i, c)| !used[*i] && c.spark >= threat.spark)
            .min_by_key(|(_, c)| c.spark)
            .map(|(i, _)| i);

        if let Some(idx) = best_idx {
            used[idx] = true;
            placements.push((
                eligible[idx].id,
                CharacterPlacement::MoveToFrontRank(threat.column),
            ));
        }
    }

    if placements.is_empty() {
        return None;
    }

    // Everyone else stays back
    for (i, c) in eligible.iter().enumerate() {
        if !used[i] {
            placements.push((c.id, CharacterPlacement::StayBack));
        }
    }

    Some(PositionAssignment { placements })
}

/// Assign weakest characters to block strongest opponents (chump blocking).
fn chump_blocking(
    eligible: &[EligibleCharacter],
    threats: &[OpponentThreat],
) -> Option<PositionAssignment> {
    if threats.is_empty() {
        return None;
    }

    let mut sorted_threats: Vec<&OpponentThreat> = threats.iter().collect();
    sorted_threats.sort_by(|a, b| b.spark.cmp(&a.spark));

    // Sort eligible by spark ascending (weakest first for chump blocking)
    let mut sorted_eligible: Vec<(usize, &EligibleCharacter)> =
        eligible.iter().enumerate().collect();
    sorted_eligible.sort_by_key(|(_, c)| c.spark);

    let mut used = vec![false; eligible.len()];
    let mut placements: Vec<(CharacterId, CharacterPlacement)> = Vec::new();

    // Only chump-block threats that would actually beat us
    for threat in &sorted_threats {
        let best_idx = sorted_eligible
            .iter()
            .find(|(i, c)| !used[*i] && c.spark < threat.spark)
            .map(|(i, _)| *i);

        if let Some(idx) = best_idx {
            used[idx] = true;
            placements.push((
                eligible[idx].id,
                CharacterPlacement::MoveToFrontRank(threat.column),
            ));
        }
    }

    if placements.is_empty() {
        return None;
    }

    for (i, c) in eligible.iter().enumerate() {
        if !used[i] {
            placements.push((c.id, CharacterPlacement::StayBack));
        }
    }

    Some(PositionAssignment { placements })
}

/// Chump-block as many threats as possible with weakest characters.
fn chump_block_all(
    eligible: &[EligibleCharacter],
    threats: &[OpponentThreat],
) -> Option<PositionAssignment> {
    if threats.is_empty() {
        return None;
    }

    let mut sorted_threats: Vec<&OpponentThreat> = threats.iter().collect();
    sorted_threats.sort_by(|a, b| b.spark.cmp(&a.spark));

    let mut sorted_eligible: Vec<(usize, &EligibleCharacter)> =
        eligible.iter().enumerate().collect();
    sorted_eligible.sort_by_key(|(_, c)| c.spark);

    let mut used = vec![false; eligible.len()];
    let mut placements: Vec<(CharacterId, CharacterPlacement)> = Vec::new();

    for threat in &sorted_threats {
        // Use any unused character (weakest first)
        let best_idx = sorted_eligible.iter().find(|(i, _)| !used[*i]).map(|(i, _)| *i);

        if let Some(idx) = best_idx {
            used[idx] = true;
            placements.push((
                eligible[idx].id,
                CharacterPlacement::MoveToFrontRank(threat.column),
            ));
        }
    }

    if placements.is_empty() {
        return None;
    }

    for (i, c) in eligible.iter().enumerate() {
        if !used[i] {
            placements.push((c.id, CharacterPlacement::StayBack));
        }
    }

    Some(PositionAssignment { placements })
}

/// Attack with characters that the opponent likely cannot profitably block.
fn attack_focused(
    eligible: &[EligibleCharacter],
    attack_col: u8,
    opp_back_max_spark: u32,
) -> Option<PositionAssignment> {
    // Find the strongest eligible character that exceeds opponent's
    // back-rank max spark (profitable attack)
    let best_attacker = eligible.iter().max_by_key(|c| c.spark)?;
    if best_attacker.spark <= opp_back_max_spark {
        return None; // No profitable attack available
    }

    let mut placements: Vec<(CharacterId, CharacterPlacement)> = Vec::new();
    for c in eligible {
        if c.id == best_attacker.id {
            placements.push((c.id, CharacterPlacement::MoveToFrontRank(attack_col)));
        } else {
            placements.push((c.id, CharacterPlacement::StayBack));
        }
    }

    Some(PositionAssignment { placements })
}

/// Block highest threat + attack with best remaining if profitable.
fn mixed_block_and_attack(
    eligible: &[EligibleCharacter],
    threats: &[OpponentThreat],
    attack_column: Option<u8>,
    opp_back_max_spark: u32,
) -> Option<PositionAssignment> {
    let highest_threat = threats.iter().max_by_key(|t| t.spark)?;

    // Find cheapest blocker that beats highest threat
    let blocker_idx = eligible
        .iter()
        .enumerate()
        .filter(|(_, c)| c.spark >= highest_threat.spark)
        .min_by_key(|(_, c)| c.spark)
        .map(|(i, _)| i);

    // If no efficient blocker, use cheapest character as chump blocker
    let blocker_idx = blocker_idx.or_else(|| {
        eligible
            .iter()
            .enumerate()
            .min_by_key(|(_, c)| c.spark)
            .map(|(i, _)| i)
    })?;

    let mut placements: Vec<(CharacterId, CharacterPlacement)> = Vec::new();
    placements.push((
        eligible[blocker_idx].id,
        CharacterPlacement::MoveToFrontRank(highest_threat.column),
    ));

    // Try to find a profitable attacker from remaining
    let mut attacked = false;
    if let Some(attack_col) = attack_column {
        let best_attacker = eligible
            .iter()
            .enumerate()
            .filter(|(i, c)| *i != blocker_idx && c.spark > opp_back_max_spark)
            .max_by_key(|(_, c)| c.spark);

        if let Some((idx, _)) = best_attacker {
            placements.push((
                eligible[idx].id,
                CharacterPlacement::MoveToFrontRank(attack_col),
            ));
            attacked = true;

            for (i, c) in eligible.iter().enumerate() {
                if i != blocker_idx && i != idx {
                    placements.push((c.id, CharacterPlacement::StayBack));
                }
            }
        }
    }

    if !attacked {
        for (i, c) in eligible.iter().enumerate() {
            if i != blocker_idx {
                placements.push((c.id, CharacterPlacement::StayBack));
            }
        }
    }

    Some(PositionAssignment { placements })
}

/// Checks if two assignments produce the same placements.
fn same_placements(a: &PositionAssignment, b: &PositionAssignment) -> bool {
    if a.placements.len() != b.placements.len() {
        return false;
    }
    a.placements.iter().all(|(id, place)| {
        b.placements.iter().any(|(bid, bplace)| id == bid && place == bplace)
    })
}

fn eligible_characters(battle: &BattleState, player: PlayerName) -> Vec<EligibleCharacter> {
    let bf = battle.cards.battlefield(player);
    let current_turn = battle.turn.turn_id.0;
    bf.back
        .iter()
        .flatten()
        .copied()
        .filter(|character_id| {
            let has_summoning_sickness = battle
                .cards
                .battlefield_state(player)
                .get(character_id)
                .is_some_and(|state| state.played_turn == current_turn);
            !has_summoning_sickness && !battle.turn.moved_this_turn.contains(character_id)
        })
        .map(|id| EligibleCharacter {
            id,
            spark: battle.cards.spark(player, id).unwrap_or_default().0,
        })
        .collect()
}

fn opponent_front_threats(battle: &BattleState, opponent: PlayerName) -> Vec<OpponentThreat> {
    battle
        .cards
        .battlefield(opponent)
        .front
        .iter()
        .enumerate()
        .filter_map(|(col, slot)| {
            slot.map(|id| OpponentThreat {
                column: col as u8,
                spark: battle.cards.spark(opponent, id).unwrap_or_default().0,
            })
        })
        .collect()
}

fn opponent_back_max_spark(battle: &BattleState, opponent: PlayerName) -> u32 {
    battle
        .cards
        .battlefield(opponent)
        .back
        .iter()
        .filter_map(|slot| *slot)
        .filter_map(|id| battle.cards.spark(opponent, id))
        .map(|s| s.0)
        .max()
        .unwrap_or(0)
}

fn find_attack_column(battle: &BattleState, player: PlayerName) -> Option<u8> {
    let opponent_front = &battle.cards.battlefield(player.opponent()).front;
    let own_front = &battle.cards.battlefield(player).front;
    opponent_front
        .iter()
        .zip(own_front.iter())
        .position(|(opp, own)| opp.is_none() && own.is_none())
        .map(|col| col as u8)
}

/// Generates a human-readable label for a position assignment at log
/// time. Only called when logging is enabled.
pub fn describe(
    battle: &BattleState,
    player: PlayerName,
    assignment: &PositionAssignment,
) -> String {
    let opponent = player.opponent();
    let mut parts = Vec::new();

    let all_stay_back = assignment
        .placements
        .iter()
        .all(|(_, p)| matches!(p, CharacterPlacement::StayBack));
    if all_stay_back {
        return "hold-all".to_string();
    }

    for &(char_id, placement) in &assignment.placements {
        match placement {
            CharacterPlacement::StayBack => {}
            CharacterPlacement::MoveToFrontRank(col) => {
                let own_spark = battle.cards.spark(player, char_id).unwrap_or_default().0;
                let opp_char = battle.cards.battlefield(opponent).front[col as usize];
                if let Some(opp_id) = opp_char {
                    let opp_spark =
                        battle.cards.spark(opponent, opp_id).unwrap_or_default().0;
                    if own_spark >= opp_spark {
                        parts.push(format!("block-{opp_spark}@col{col}(with-{own_spark})"));
                    } else {
                        parts.push(format!("chump-{own_spark}@col{col}(vs-{opp_spark})"));
                    }
                } else {
                    parts.push(format!("attack-{own_spark}@col{col}"));
                }
            }
        }
    }

    if parts.is_empty() {
        "hold-all".to_string()
    } else {
        parts.join("+")
    }
}
```

```rust
use battle_mutations::actions::apply_battle_action;
use battle_mutations::phase_mutations::turn;
use battle_queries::legal_action_queries::legal_actions;
use battle_queries::legal_action_queries::legal_actions_data::LegalActions;
use battle_state::actions::battle_actions::BattleAction;
use battle_state::battle::battle_state::BattleState;

use core_data::types::PlayerName;

use crate::position_assignment::{self, CharacterPlacement, PositionAssignment};

/// Atomically applies a position assignment to the battle state.
///
/// Executes BeginPositioning, then SelectCharacterForPositioning +
/// MoveCharacterToFrontRank for each character with a front-rank
/// placement, then EndTurn to transition through ending and judgment.
pub fn apply_position_assignment(
    battle: &mut BattleState,
    player: PlayerName,
    assignment: &PositionAssignment,
) {
    apply_battle_action::execute(battle, player, BattleAction::BeginPositioning);

    for &(char_id, placement) in &assignment.placements {
        if let CharacterPlacement::MoveToFrontRank(col) = placement {
            apply_battle_action::execute(
                battle,
                player,
                BattleAction::SelectCharacterForPositioning(char_id),
            );
            apply_battle_action::execute(
                battle,
                player,
                BattleAction::MoveCharacterToFrontRank(char_id, col),
            );
        }
    }

    apply_battle_action::execute(battle, player, BattleAction::EndTurn);
}

/// Plays a greedy card-play phase: repeatedly play the highest-cost
/// affordable card until energy is exhausted.
fn greedy_card_play(battle: &mut BattleState, player: PlayerName) {
    loop {
        let legal = legal_actions::compute(battle, player);
        match &legal {
            LegalActions::Standard { actions } => {
                // Find highest-cost playable card from hand
                let best_card = actions
                    .play_card_from_hand
                    .iter()
                    .max_by_key(|hand_card_id| {
                        battle.cards.energy_cost(player, *hand_card_id).unwrap_or_default().0
                    });

                if let Some(card_id) = best_card {
                    let action = BattleAction::PlayCardFromHand(card_id);
                    apply_battle_action::execute(battle, player, action);

                    // After playing a card, resolve any prompts
                    // and stack interactions with simple defaults
                    resolve_pending_with_defaults(battle, player);
                    continue;
                }

                // Try highest-cost card from void
                let best_void = actions
                    .play_card_from_void
                    .iter()
                    .max_by_key(|void_card_id| {
                        battle.cards.void_energy_cost(player, *void_card_id).unwrap_or_default().0
                    });

                if let Some(card_id) = best_void {
                    let action = BattleAction::PlayCardFromVoid(card_id);
                    apply_battle_action::execute(battle, player, action);
                    resolve_pending_with_defaults(battle, player);
                    continue;
                }

                // Try activating abilities
                if let Some(char_id) = actions.activate_abilities_for_character.iter().next() {
                    let action = BattleAction::ActivateAbilityForCharacter(char_id);
                    apply_battle_action::execute(battle, player, action);
                    resolve_pending_with_defaults(battle, player);
                    continue;
                }

                // Nothing left to play
                break;
            }
            _ => {
                // We're in a prompt or non-standard state — resolve it
                resolve_pending_with_defaults(battle, player);
                if matches!(legal, LegalActions::Standard { .. }) {
                    continue;
                }
                break;
            }
        }
    }
}

/// Resolve any pending prompts/stack with default choices.
fn resolve_pending_with_defaults(battle: &mut BattleState, player: PlayerName) {
    loop {
        let Some(acting_player) = legal_actions::next_to_act(battle) else {
            break;
        };

        let legal = legal_actions::compute(battle, acting_player);
        match &legal {
            LegalActions::Standard { .. } if acting_player == player => break,
            _ => {
                if let Some(action) = legal.random_action() {
                    apply_battle_action::execute(battle, acting_player, action);
                } else {
                    break;
                }
            }
        }
    }
}

/// Plays out a complete turn for the given player using greedy
/// heuristics. Used during rollouts for both AI and opponent turns.
///
/// Assumes the player's turn is currently in the Main phase.
/// After this function returns, the game state will have advanced
/// through the turn's ending and judgment phases.
pub fn play_greedy_turn(battle: &mut BattleState, player: PlayerName) {
    // Card play phase
    greedy_card_play(battle, player);

    // Positioning phase
    let legal = legal_actions::compute(battle, player);
    if let LegalActions::Standard { actions } = &legal {
        if actions.can_begin_positioning {
            if let Some(assignment) = position_assignment::best_assignment(battle, player) {
                apply_position_assignment(battle, player, &assignment);
                return;
            }
        }
    }

    // End turn (no positioning or hold-all was best)
    apply_battle_action::execute(battle, player, BattleAction::EndTurn);
}
```

```rust
/// Represents either a standard BattleAction or a position assignment
/// that the V2 search evaluates as a root-level candidate.
#[derive(Debug, Clone)]
enum SearchCandidate {
    Action(BattleAction),
    Assignment(PositionAssignment),
}

/// Monte Carlo search with heuristic position assignments.
pub fn search(
    initial_battle: &BattleState,
    player: PlayerName,
    config: &UctConfig,
) -> BattleAction {
    let legal = legal_actions::compute(initial_battle, player);
    let candidates = build_candidates(initial_battle, player, &legal);
    let budget = compute_budget(&legal, config, initial_battle, player, candidates.len());

    let action_results: Vec<_> = candidates
        .par_iter()
        .with_min_len(if config.single_threaded { usize::MAX } else { 1 })
        .map(|candidate| {
            search_candidate(
                initial_battle,
                player,
                budget.iterations_per_action,
                candidate,
                None,
            )
        })
        .collect();

    let Some(best_result) = action_results.iter().max_by_key(|result| {
        if result.visit_count == 0 {
            OrderedFloat(-f64::INFINITY)
        } else {
            OrderedFloat(result.total_reward.0 / result.visit_count as f64)
        }
    }) else {
        panic_with!("No legal actions available", initial_battle, player);
    };

    let action = resolve_action(best_result);
    let num_actions = candidates.len();
    let total_iterations = budget.iterations_per_action * num_actions as u32;
    let num_threads = rayon::current_num_threads();

    debug!(?total_iterations, ?action, ?num_threads, "Picked V2 AI action");

    if initial_battle.request_context.logging_options.log_ai_decisions {
        write_decision_log_entry(
            initial_battle,
            player,
            &action_results,
            best_result,
            &budget,
            num_actions,
            num_threads,
        );
    }

    action
}

/// Builds the list of root-level candidates to evaluate.
///
/// Standard actions (card plays, EndTurn, etc.) are included directly.
/// If positioning is available, BeginPositioning is replaced with
/// heuristic-generated position assignments.
fn build_candidates(
    battle: &BattleState,
    player: PlayerName,
    legal: &LegalActions,
) -> Vec<SearchCandidate> {
    let all_actions = legal.all();
    let mut candidates: Vec<SearchCandidate> = Vec::new();

    let has_positioning = all_actions.contains(&BattleAction::BeginPositioning);

    for action in &all_actions {
        if *action == BattleAction::BeginPositioning {
            continue; // Replaced by assignments below
        }
        candidates.push(SearchCandidate::Action(*action));
    }

    if has_positioning {
        let assignments = position_assignment::generate(battle, player);
        for assignment in assignments {
            candidates.push(SearchCandidate::Assignment(assignment));
        }
    }

    candidates
}

/// Resolves a search result to the BattleAction to return.
///
/// For standard actions, returns the action directly.
/// For position assignments, returns BeginPositioning as the first
/// action. The AI controller will then emit the full sequence of
/// SelectCharacterForPositioning + MoveCharacterToFrontRank actions
/// by storing the winning assignment and stepping through it.
fn resolve_action(result: &CandidateSearchResult) -> BattleAction {
    match &result.candidate {
        SearchCandidate::Action(action) => *action,
        SearchCandidate::Assignment(_) => BattleAction::BeginPositioning,
    }
}

fn search_candidate(
    initial_battle: &BattleState,
    player: PlayerName,
    iterations_per_action: u32,
    candidate: &SearchCandidate,
    randomize_player_seed: Option<u64>,
) -> CandidateSearchResult {
    let subscriber = tracing_subscriber::registry().with(EnvFilter::new("warn"));
    tracing::subscriber::with_default(subscriber, || {
        let mut graph = SearchGraph::default();
        let root = graph.add_node(SearchNode {
            player,
            total_reward: OrderedFloat(0.0),
            visit_count: 0,
            tried: Vec::new(),
        });

        let mut randomize_player_rng = Xoshiro256PlusPlus::seed_from_u64(
            randomize_player_seed.unwrap_or_else(|| rand::rng().random()),
        );

        let mut wins = 0u32;
        let mut losses = 0u32;
        let mut draws = 0u32;
        let log_tree_stats = initial_battle.request_context.logging_options.log_ai_decisions;
        let mut traversal_stats =
            if log_tree_stats { Some(TreeTraversalAccumulator::default()) } else { None };

        for _ in 0..iterations_per_action {
            let mut battle = player_state::randomize_battle_player(
                initial_battle,
                player.opponent(),
                randomize_player_rng.random(),
            );
            battle.request_context.logging_options.enable_action_legality_check = false;

            // Apply the root candidate
            apply_candidate(&mut battle, player, candidate);

            let node = next_evaluation_target(
                &mut battle,
                &mut graph,
                root,
                player,
                traversal_stats.as_mut(),
            );
            let reward = evaluate(&mut battle, player);
            back_propagate_rewards(&mut graph, player, node, reward);

            match reward.0 {
                r if r > 0.0 => wins += 1,
                r if r < 0.0 => losses += 1,
                _ => draws += 1,
            }
        }

        let total_reward = graph[root].total_reward;
        let visit_count = graph[root].visit_count;
        let tree_node_count = graph.node_count();
        let tree_max_depth = decision_log::compute_tree_depth(&graph, root);
        let depth_stats = traversal_stats.map(TreeTraversalAccumulator::into_depth_stats);
        CandidateSearchResult {
            candidate: candidate.clone(),
            graph,
            root,
            total_reward,
            visit_count,
            wins,
            losses,
            draws,
            tree_node_count,
            tree_max_depth,
            depth_stats,
        }
    })
}

/// Applies a SearchCandidate to the battle state.
fn apply_candidate(
    battle: &mut BattleState,
    player: PlayerName,
    candidate: &SearchCandidate,
) {
    battle.request_context.logging_options.enable_action_legality_check = false;
    match candidate {
        SearchCandidate::Action(action) => {
            apply_battle_action::execute(battle, player, *action);
        }
        SearchCandidate::Assignment(assignment) => {
            rollout_policy::apply_position_assignment(battle, player, assignment);
        }
    }
}

struct CandidateSearchResult {
    candidate: SearchCandidate,
    graph: SearchGraph,
    root: NodeIndex,
    total_reward: OrderedFloat<f64>,
    visit_count: u32,
    wins: u32,
    losses: u32,
    draws: u32,
    tree_node_count: usize,
    tree_max_depth: u32,
    depth_stats: Option<Vec<DepthLevelStats>>,
}

/// Tree policy: find an unexplored node or the best terminal descendant.
///
/// Key difference from V1: opponent decision points are collapsed into
/// a greedy heuristic instead of expanded as tree nodes. Only AI
/// decision points (including fast-card responses during opponent turns)
/// branch in the tree.
fn next_evaluation_target(
    battle: &mut BattleState,
    graph: &mut SearchGraph,
    from_node: NodeIndex,
    maximizing_player: PlayerName,
    mut stats: Option<&mut TreeTraversalAccumulator>,
) -> NodeIndex {
    let mut node = from_node;
    let mut depth = 0usize;
    while let Some(player) = legal_actions::next_to_act(battle) {
        // Collapse opponent decisions (except when AI has fast-card response)
        if player != maximizing_player {
            play_opponent_turn_heuristic(battle, player);
            continue;
        }

        let actions = legal_actions::compute(battle, player);
        let explored = &graph[node].tried;

        if let Some(action) = actions.find_missing(explored) {
            if let Some(s) = stats.as_mut() {
                s.record_expansion(depth, player, &action);
            }
            return add_child(battle, graph, player, node, action);
        } else {
            if let Some(s) = stats.as_mut() {
                s.record_selection(depth, player);
            }
            let best = best_child(graph, node, &actions, SelectionMode::Exploration);
            battle.request_context.logging_options.enable_action_legality_check = false;
            apply_battle_action::execute(battle, player, best.action);
            node = best.node;
        }
        depth += 1;
    }
    node
}

/// Plays out the opponent's remaining decisions using greedy heuristics.
fn play_opponent_turn_heuristic(battle: &mut BattleState, opponent: PlayerName) {
    loop {
        let Some(player) = legal_actions::next_to_act(battle) else {
            break;
        };
        if player != opponent {
            break; // AI gets to act (e.g., fast card response)
        }
        let legal = legal_actions::compute(battle, player);
        match &legal {
            LegalActions::Standard { actions } => {
                // Greedy: play highest-cost card
                let card_action = actions
                    .play_card_from_hand
                    .iter()
                    .max_by_key(|id| {
                        battle.cards.energy_cost(player, *id).unwrap_or_default().0
                    })
                    .map(BattleAction::PlayCardFromHand);

                if let Some(action) = card_action {
                    apply_battle_action::execute(battle, player, action);
                    continue;
                }

                // Positioning
                if actions.can_begin_positioning {
                    if let Some(assignment) =
                        position_assignment::best_assignment(battle, player)
                    {
                        rollout_policy::apply_position_assignment(
                            battle, player, &assignment,
                        );
                        continue;
                    }
                }

                // End turn / pass
                let primary_action = match actions.primary {
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::EndTurn => BattleAction::EndTurn,
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::PassPriority => BattleAction::PassPriority,
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::StartNextTurn => BattleAction::StartNextTurn,
                };
                apply_battle_action::execute(battle, player, primary_action);
            }
            _ => {
                // Resolve prompts with random choice
                if let Some(action) = legal.random_action() {
                    apply_battle_action::execute(battle, player, action);
                } else {
                    break;
                }
            }
        }
    }
}

fn add_child(
    battle: &mut BattleState,
    graph: &mut SearchGraph,
    player: PlayerName,
    parent: NodeIndex,
    action: BattleAction,
) -> NodeIndex {
    battle.request_context.logging_options.enable_action_legality_check = false;
    graph[parent].tried.push(action);
    apply_battle_action::execute(battle, player, action);
    let child = graph.add_node(SearchNode {
        player,
        total_reward: OrderedFloat(0.0),
        visit_count: 0,
        tried: Vec::new(),
    });
    graph.add_edge(parent, child, SearchEdge { action });
    child
}

struct BestChild {
    action: BattleAction,
    node: NodeIndex,
}

fn best_child(
    graph: &SearchGraph,
    from_node: NodeIndex,
    legal: &LegalActions,
    selection_mode: SelectionMode,
) -> BestChild {
    let parent_visits = graph[from_node].visit_count;

    graph
        .edges(from_node)
        .filter(|e| legal.contains(e.weight().action, ForPlayer::Agent))
        .max_by_key(|edge| {
            let target = edge.target();
            child_score(
                parent_visits,
                graph[target].visit_count,
                graph[target].total_reward,
                selection_mode,
            )
        })
        .map(|edge| BestChild { action: edge.weight().action, node: edge.target() })
        .expect("No legal children available")
}

fn back_propagate_rewards(
    graph: &mut SearchGraph,
    maximizing_player: PlayerName,
    leaf_node: NodeIndex,
    reward: OrderedFloat<f64>,
) {
    let mut node = leaf_node;
    loop {
        let weight = &mut graph[node];
        weight.visit_count += 1;
        weight.total_reward += if weight.player == maximizing_player { reward } else { -reward };

        node = match graph.neighbors_directed(node, Direction::Incoming).next() {
            Some(n) => n,
            _ => break,
        };
    }
}

/// Rollout: play greedy turns until the game ends.
fn evaluate(battle: &mut BattleState, maximizing_player: PlayerName) -> OrderedFloat<f64> {
    while let Some(player) = legal_actions::next_to_act(battle) {
        let legal = legal_actions::compute(battle, player);
        match &legal {
            LegalActions::Standard { actions } if actions.can_begin_positioning => {
                // Use atomic heuristic positioning
                if let Some(assignment) = position_assignment::best_assignment(battle, player)
                {
                    rollout_policy::apply_position_assignment(battle, player, &assignment);
                    continue;
                }
                // Fallback: end turn
                apply_battle_action::execute(battle, player, BattleAction::EndTurn);
            }
            LegalActions::Standard { actions } => {
                // Greedy card play
                let card_action = actions
                    .play_card_from_hand
                    .iter()
                    .max_by_key(|id| {
                        battle.cards.energy_cost(player, *id).unwrap_or_default().0
                    })
                    .map(BattleAction::PlayCardFromHand);

                if let Some(action) = card_action {
                    apply_battle_action::execute(battle, player, action);
                    continue;
                }

                // No cards to play, end turn or pass
                let primary_action = match actions.primary {
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::EndTurn => BattleAction::EndTurn,
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::PassPriority => BattleAction::PassPriority,
                    battle_queries::legal_action_queries::legal_actions_data::PrimaryLegalAction::StartNextTurn => BattleAction::StartNextTurn,
                };
                apply_battle_action::execute(battle, player, primary_action);
            }
            _ => {
                // Prompts, positioning sub-steps, etc. — resolve randomly
                if let Some(action) = legal.random_action() {
                    apply_battle_action::execute(battle, player, action);
                } else {
                    break;
                }
            }
        }
    }

    let BattleStatus::GameOver { winner } = battle.status else {
        panic_with!("Battle has not ended", battle);
    };
    let reward = if winner == Some(maximizing_player) {
        1.0
    } else if winner.is_some() {
        -1.0
    } else {
        0.0
    };
    OrderedFloat(reward)
}

fn child_score(
    parent_visits: u32,
    child_visits: u32,
    reward: OrderedFloat<f64>,
    selection_mode: SelectionMode,
) -> OrderedFloat<f64> {
    let exploitation = reward / f64::from(child_visits);
    let exploration =
        f64::sqrt((2.0 * f64::ln(f64::from(parent_visits))) / f64::from(child_visits));
    let exploration_bias = match selection_mode {
        SelectionMode::Exploration => consts::FRAC_1_SQRT_2,
        SelectionMode::RewardOnly => 0.0,
    };
    exploitation + (exploration_bias * exploration)
}

struct BudgetInfo {
    iterations_per_action: u32,
    base_iterations: u32,
    multiplier: f64,
    multiplier_reason: &'static str,
}

fn compute_budget(
    legal: &LegalActions,
    config: &UctConfig,
    battle: &BattleState,
    agent: PlayerName,
    candidate_count: usize,
) -> BudgetInfo {
    let base_iterations = match candidate_count {
        0 => config.max_iterations_per_action,
        action_count => {
            let total_budget =
                config.max_iterations_per_action * config.max_total_actions_multiplier;
            let distributed_iterations = (total_budget as f64 / action_count as f64) as u32;
            cmp::min(distributed_iterations, config.max_iterations_per_action)
        }
    };

    let is_main =
        battle.turn.active_player == agent && matches!(battle.phase, BattleTurnPhase::Main);
    let player_state = battle.players.player(battle.turn.active_player);

    let (multiplier, multiplier_reason) = match is_main {
        _ if legal.is_prompt() => (0.5, "prompt"),
        true if player_state.current_energy >= player_state.produced_energy => {
            (1.5, "first_main")
        }
        true => (1.0, "main"),
        _ => (0.75, "other"),
    };
    let (applied_multiplier, applied_reason) = match config.iteration_multiplier_override {
        Some(m) => (m, "override"),
        None => (multiplier, multiplier_reason),
    };
    BudgetInfo {
        iterations_per_action: ((base_iterations as f64) * applied_multiplier) as u32,
        base_iterations,
        multiplier: applied_multiplier,
        multiplier_reason: applied_reason,
    }
}

fn write_decision_log_entry(
    battle: &BattleState,
    player: PlayerName,
    action_results: &[CandidateSearchResult],
    best_result: &CandidateSearchResult,
    budget: &BudgetInfo,
    num_actions: usize,
    num_threads: usize,
) {
    let best_avg = if best_result.visit_count == 0 {
        0.0
    } else {
        best_result.total_reward.0 / best_result.visit_count as f64
    };

    let chosen_action_str = match &best_result.candidate {
        SearchCandidate::Action(a) => format!("{a:?}"),
        SearchCandidate::Assignment(a) => {
            format!("PositionAssignment({})", position_assignment::describe(battle, player, a))
        }
    };

    let chosen_action_short = match &best_result.candidate {
        SearchCandidate::Action(a) => a.battle_action_string(),
        SearchCandidate::Assignment(a) => position_assignment::describe(battle, player, a),
    };

    let mut results: Vec<ActionResult> = action_results
        .iter()
        .map(|r| {
            let avg =
                if r.visit_count == 0 { 0.0 } else { r.total_reward.0 / r.visit_count as f64 };
            let (action_str, short_str) = match &r.candidate {
                SearchCandidate::Action(a) => (format!("{a:?}"), a.battle_action_string()),
                SearchCandidate::Assignment(a) => (
                    format!(
                        "PositionAssignment({})",
                        position_assignment::describe(battle, player, a)
                    ),
                    position_assignment::describe(battle, player, a),
                ),
            };
            let is_chosen = match (&r.candidate, &best_result.candidate) {
                (SearchCandidate::Action(a), SearchCandidate::Action(b)) => a == b,
                (SearchCandidate::Assignment(_), SearchCandidate::Assignment(_)) => {
                    std::ptr::eq(r, best_result)
                }
                _ => false,
            };
            ActionResult {
                action: action_str,
                action_short: short_str,
                total_reward: r.total_reward.0,
                visit_count: r.visit_count,
                avg_reward: avg,
                wins: r.wins,
                losses: r.losses,
                draws: r.draws,
                tree_node_count: r.tree_node_count,
                tree_max_depth: r.tree_max_depth,
                depth_stats: if is_chosen { r.depth_stats.clone() } else { None },
            }
        })
        .collect();
    results.sort_by(|a, b| b.avg_reward.partial_cmp(&a.avg_reward).unwrap_or(Ordering::Equal));

    let entry = DecisionLogEntry {
        timestamp: Utc::now().to_rfc3339(),
        player: format!("{player:?}"),
        chosen_action: chosen_action_str,
        chosen_action_short,
        chosen_avg_reward: best_avg,
        game_state: decision_log::build_game_state_snapshot(battle),
        budget: BudgetDetails {
            iterations_per_action: budget.iterations_per_action,
            base_iterations: budget.base_iterations,
            total_iterations: budget.iterations_per_action * num_actions as u32,
            num_actions,
            multiplier: budget.multiplier,
            multiplier_reason: budget.multiplier_reason.to_string(),
            num_threads,
        },
        action_results: results,
    };
    decision_log::write_decision_log(&entry, &battle.request_context);
```

```rust
/// If there is a pending position assignment, returns the next action
/// in the sequence. Returns None if no assignment is pending or the
/// assignment is complete.
fn next_assignment_action(battle: &BattleState, player: PlayerName) -> Option<BattleAction> {
    PENDING_ASSIGNMENT.with(|cell| {
        let mut assignment = cell.borrow_mut();
        let Some(ref mut a) = *assignment else {
            return None;
        };

        let legal = legal_actions::compute(battle, player);

        match &legal {
            LegalActions::SelectPositioningCharacter { .. } => {
                // Find first character that needs to move to front and
                // is still in back rank
                for &(char_id, placement) in &a.placements {
                    if let CharacterPlacement::MoveToFrontRank(_) = placement {
                        if battle.cards.battlefield(player).is_in_back_rank(char_id) {
                            return Some(BattleAction::SelectCharacterForPositioning(char_id));
                        }
                    }
                }

                // All characters positioned, end turn
                *assignment = None;
                Some(BattleAction::EndTurn)
            }
            LegalActions::AssignColumn { character, .. } => {
                // Find the column for this character
                for &(char_id, placement) in &a.placements {
                    if char_id == *character {
                        if let CharacterPlacement::MoveToFrontRank(col) = placement {
                            return Some(BattleAction::MoveCharacterToFrontRank(char_id, col));
                        }
                    }
                }
                // Shouldn't happen, but fallback
                *assignment = None;
                None
            }
            _ => {
                // Not in positioning flow, clear assignment
                *assignment = None;
                None
            }
        }
    })
}
```

```rust
    if let Some(action) = next_assignment_action(battle, player) {

        max_iterations_per_action: *thousands_of_iterations * 1000,
        max_total_actions_multiplier: 6,
        iteration_multiplier_override,
        single_threaded: false,
    };
    let result = uct_search_v2::search(battle, player, &config);
    if let Some(assignment) = result.assignment {
        PENDING_ASSIGNMENT.with(|cell| {
            *cell.borrow_mut() = Some(assignment);
        });
    }
    result.action
}
```
